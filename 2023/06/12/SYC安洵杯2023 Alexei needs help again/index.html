class A:
    def __init__(self,s,m):
        self.s=s
        self.max=m//s
        self.n=m%s
        self.L=lucas(self.s)
    def nn(self,i):
        return fibonacci(self.n+i*self.s)
    def ab(self):
        return self.L,(-1)^(self.s+1)
    def sumall(self):
        a,b=self.ab()
        M=Matrix(ZZ,[[a,b,0],[1,0,0],[1,0,1]])
        S=vector(ZZ,[self.nn(1),self.nn(0),self.nn(0)])
        if (self.max-1)<0:
            return 0
        return (M^(self.max-1)*S)[-1]
